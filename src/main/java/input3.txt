proc { } 
func { 
    factorial (n) { 
    local { result i } 
    result = 1 ; 
    i = 1 ; 
    while (i > n) { 
        result = (result mult i) ; 
        i = (i plus 1) 
    } ; 
    return result 
    } 
} 
main { 
    var { answer } 
    answer = factorial (5) ; 
    print answer ; 
    halt
}
